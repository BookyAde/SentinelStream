PostgreSQL Schema

-- ───────────────────────────────────────────── -- TABLE: events -- Primary store for all pipeline events -- ───────────────────────────────────────────── CREATE TABLE events ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), external_id VARCHAR(255) UNIQUE, -- idempotency key event_type VARCHAR(100) NOT NULL, source VARCHAR(100) NOT NULL, priority eventpriority NOT NULL DEFAULT 'normal', payload JSONB NOT NULL, metadata JSONB, status eventstatus NOT NULL DEFAULT 'queued', retry_count INTEGER NOT NULL DEFAULT 0, error_message TEXT, processor_id VARCHAR(100), -- hostname:pid of the worker created_at TIMESTAMPTZ NOT NULL DEFAULT now(), updated_at TIMESTAMPTZ NOT NULL DEFAULT now(), processed_at TIMESTAMPTZ ); CREATE INDEX ix_events_status ON events (status); CREATE INDEX ix_events_event_type ON events (event_type); CREATE INDEX ix_events_source ON events (source); CREATE INDEX ix_events_created_at ON events (created_at); CREATE INDEX ix_events_priority_status ON events (priority, status); -- ───────────────────────────────────────────── -- TABLE: dead_letter_queue -- Persistent record of exhausted-retry events -- ───────────────────────────────────────────── CREATE TABLE dead_letter_queue ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), original_event_id UUID, -- soft FK to events event_type VARCHAR(100) NOT NULL, source VARCHAR(100) NOT NULL, payload JSONB NOT NULL, failure_reason TEXT NOT NULL, retry_count INTEGER NOT NULL DEFAULT 0, last_error TEXT, replayed BOOLEAN NOT NULL DEFAULT FALSE, replayed_at TIMESTAMPTZ, replay_event_id UUID, -- new event created on replay created_at TIMESTAMPTZ NOT NULL DEFAULT now(), updated_at TIMESTAMPTZ NOT NULL DEFAULT now() ); CREATE INDEX ix_dlq_replayed ON dead_letter_queue (replayed); CREATE INDEX ix_dlq_event_type ON dead_letter_queue (event_type); CREATE INDEX ix_dlq_created_at ON dead_letter_queue (created_at); -- ───────────────────────────────────────────── -- ENUMS -- ───────────────────────────────────────────── CREATE TYPE eventstatus AS ENUM ( 'queued', 'processing', 'processed', 'failed', 'replayed' ); CREATE TYPE eventpriority AS ENUM ( 'low', 'normal', 'high', 'critical' );

Technology Decisions

Queue Transport
Redis LIST (RPUSH/LPOP) — O(1), ordered, in-memory speed with AOF persistence
Dead Letter Queue
Dual store: Redis list for fast inspection + PostgreSQL for audit trail and replay metadata
Idempotency
external_id UNIQUE index prevents duplicate ingestion; returns status=duplicate on conflict
Retry Strategy
Exponential backoff (2ⁿ seconds), max 3 attempts, configurable via RETRY_BACKOFF_BASE env var
Concurrency
N worker replicas via Docker Compose deploy.replicas; each drains batches of 50 events asynchronously
Observability
Structured JSON logs with correlation IDs; /health for k8s probes; /metrics for counters and throughput
